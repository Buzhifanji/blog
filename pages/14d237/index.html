# all：布尔全等判断

const all = (arr, fn = Boolean) => arr.every(fn);
1
all([4, 2, 3], x => x > 1); // true
all([1, 2, 3]); // true
1

# allEqual：检查数组各项相等

const allEqual = arr => arr.every(val => val === arr[0]);
1
allEqual([1, 2, 3, 4, 5, 6]); // false
allEqual([1, 1, 1, 1]); // true
1
2

# allEqualBy：检查数组各项是否满足条件（四舍五入）

const allEqualBy = (arr, fn) => {
  const eql = fn(arr[0]);
  return arr.every(val => fn(val) === eql);
};
1
2
3
4
allEqualBy([1.1, 1.2, 1.3], Math.round); // true
allEqualBy([1.1, 1.3, 1.6], Math.round); // false
1
2

# allUnique：检查数组是否重复

const allUnique = arr => arr.length === new Set(arr).size;
1
allUnique([1, 2, 3, 4]); // true
allUnique([1, 1, 2, 3]); // false
1

# allUniqueBy：根据条件（四舍五入）检查数组是否重复

const allUniqueBy = (arr, fn) => arr.length === new Set(arr.map(fn)).size;
1
allUniqueBy([1.2, 2.4, 2.9], Math.round); // true
allUniqueBy([1.2, 2.3, 2.4], Math.round); // false

# aperture：分割数组，每份长度为n

const aperture = (n, arr) =>
  n > arr.length
    ? []
    : arr.slice(n - 1).map((v, i) => arr.slice(i, i + n));
1
2
3
4
aperture(2, [1, 2, 3, 4]); // [[1, 2], [2, 3], [3, 4]]
aperture(3, [1, 2, 3, 4]); // [[1, 2, 3], [2, 3, 4]]
aperture(5, [1, 2, 3, 4]); // []

# approximatelyEqual：约等于

const approximatelyEqual = (v1, v2, epsilon = 0.001) =>
  Math.abs(v1 - v2) < epsilon;
1
2

# arithmeticProgression：等差数列

const arithmeticProgression  = (n, lim) =>
  Array.from({ length: Math.ceil(lim / n) }, (_, i) => (i + 1) * n );
1
2
arithmeticProgression(5, 25); // [5, 10, 15, 20, 25]

# arrayToCSV：数组转CSV格式（带空格的字符串）

const arrayToCSV = (arr, delimiter = ',') =>
  arr
    .map(v =>
      v.map(x => (isNaN(x) ? `"${x.replace(/"/g, '""')}"` : x)).join(delimiter)
    )
    .join('\n');
1
2
3
4
5
6
arrayToCSV([['a', 'b'], ['c', 'd']]); // '"a","b"\n"c","d"'
arrayToCSV([['a', 'b'], ['c', 'd']], ';'); // '"a";"b"\n"c";"d"'
arrayToCSV([['a', '"b" great'], ['c', 3.1415]]); // '"a","""b"" great"\n"c",3.1415'

此代码段将数组的元素转换为 li 标签，并将其附加到给定ID的列表中。

  document.querySelector(`#${listID}`).innerHTML += arr

# average：平均数

const average = (...nums) =>
  nums.reduce((acc, val) => acc + val, 0) / nums.length;
1
2
average(...[1, 2, 3]); // 2
average(1, 2, 3); // 2
1

# averageBy：数组对象属性平均数

const averageBy = (arr, fn) =>
  arr
    .map(typeof fn === 'function' ? fn : val => val[fn])
    .reduce((acc, val) => acc + val, 0) / arr.length;
1
2
3
4
averageBy([{ n: 4 }, { n: 2 }, { n: 8 }, { n: 6 }], o => o.n); // 5
averageBy([{ n: 4 }, { n: 2 }, { n: 8 }, { n: 6 }], 'n'); // 5

# JSONtoCSV: JSON数组转CSV格式

const JSONtoCSV = (arr, columns, delimiter = ',') =>
  [
    columns.join(delimiter),
    ...arr.map(obj =>
      columns.reduce(
        (acc, key) =>
          `${acc}${!acc.length ? '' : delimiter}"${!obj[key] ? '' : obj[key]}"`,
        ''
      )
    ),
  ].join('\n');
